```sql
SELECT
  p.proname,
  p.prosecdef,
  pg_get_function_identity_arguments(p.oid) AS args,
  array_agg(acl.privilege_type) AS grants
FROM pg_proc p
JOIN pg_namespace n ON n.oid = p.pronamespace
LEFT JOIN aclexplode(p.proacl) acl
  ON acl.grantee = (SELECT oid FROM pg_roles WHERE rolname = 'authenticated')
WHERE n.nspname = 'public' AND p.proname = 'mark_job_filled'
GROUP BY p.oid, p.proname, p.prosecdef;
```
