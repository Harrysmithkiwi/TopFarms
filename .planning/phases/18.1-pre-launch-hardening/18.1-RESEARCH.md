- **Wave 0** — RED-stub test scaffold (5 vitest files; pattern from `17-00-test-scaffold-PLAN.md`).
- **Wave 1** — Three disjoint code-only plans in parallel: `mark_job_filled` SECURITY DEFINER RPC + MarkFilledModal client refactor (#17), `match_scores` cleanup trigger + backfill (#19), 15× FK index migration (#10a). All three land via `Studio SQL Editor` per CLAUDE §2 — MCP `--read-only` stays ON.
- **Wave 2** — Two code-only plans in parallel: X-Webhook-Secret defence-in-depth across `notify-job-filled` + `send-followup-emails` Edge functions + a NEW migration `026_*` that ALTERs the two pg_net callers (#15); doc-types canonical-source vitest drift test (#1).
- **Wave 3** — Operator-action plan: Stripe live-mode posture verification (#13), `WEBHOOK_SECRET` generation + `supabase secrets set` + per-DB `ALTER DATABASE ... SET app.webhook_secret`, `supabase functions deploy` for both edited fns, then synthetic-call UAT.

**Critical sequencing for #15:** Operator MUST set both `WEBHOOK_SECRET` (Edge fn secret) AND `app.webhook_secret` (DB-level setting) BEFORE the migration runs in Wave 2 — but because the migration uses `current_setting('app.webhook_secret', true)` (true = silent NULL on missing), the migration is non-blocking; trigger fires would just send `null` header (Edge fn rejects 401, no harm). The migration can land code-first; functions break harmlessly until operator finishes Wave 3. Document this explicitly in the operator plan.

The single biggest risk: any code path that lands in production WITHOUT the corresponding operator step (e.g., Edge fn redeploy with `WEBHOOK_SECRET` validation but operator hasn't set the secret) hard-fails legitimate calls. Plans must guard with `if (!WEBHOOK_SECRET) → 503` (matches `get-resend-stats:43-46` precedent), not 401, so the failure mode is "configured" vs "invalid header."

**Primary recommendation:** Mirror the existing repo patterns 1:1. Every skeleton in this doc is derived from a specific in-repo file — no novel design surface.

### Pattern 1: SECURITY DEFINER RPC w/ caller-ownership check

**What:** Encapsulate multi-statement business logic in a Postgres function so it runs in a single implicit transaction. Bypass RLS via `SECURITY DEFINER` after explicitly validating the caller's ownership.

**When to use:** Any client-side multi-UPDATE flow where partial commit is unsafe. Phase 18.1's #17 is the canonical example — `applications.update({status:'hired'})` followed by `jobs.update({status:'filled'})` must be all-or-nothing.

**Example skeleton (full body — paste-ready):**

```sql
-- Source: derived from supabase/migrations/018_set_user_role_rpc.sql
-- (SECURITY DEFINER + caller validation + idempotent semantics).
-- 026_mark_job_filled_rpc.sql

BEGIN;

CREATE OR REPLACE FUNCTION public.mark_job_filled(
  p_job_id      uuid,
  p_applicant_id uuid
)
RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_caller_user_id uuid;
  v_owner_user_id  uuid;
  v_current_status text;
BEGIN
  -- 1. Auth — must be authenticated
  v_caller_user_id := auth.uid();
  IF v_caller_user_id IS NULL THEN
    RAISE EXCEPTION 'Not authenticated' USING ERRCODE = '42501';
  END IF;

  -- 2. Ownership — caller must own the job
  --    jobs.employer_id -> employer_profiles.id (FK)
  --    employer_profiles.user_id -> auth.users.id (1:1 unique)
  SELECT ep.user_id
    INTO v_owner_user_id
    FROM public.jobs j
    JOIN public.employer_profiles ep ON ep.id = j.employer_id
   WHERE j.id = p_job_id;

  IF v_owner_user_id IS NULL THEN
    RAISE EXCEPTION 'Job not found' USING ERRCODE = 'P0002';
  END IF;
  IF v_owner_user_id <> v_caller_user_id THEN
    RAISE EXCEPTION 'Not authorised to mark this job filled' USING ERRCODE = '42501';
  END IF;

  -- 3. Idempotency — re-marking a filled job is a no-op (RAISE NOTICE only)
  SELECT status INTO v_current_status FROM public.jobs WHERE id = p_job_id;
  IF v_current_status = 'filled' THEN
    RAISE NOTICE 'mark_job_filled: job % already filled — no-op', p_job_id;
    RETURN;
  END IF;

  -- 4. Mark applicant hired FIRST (matches MarkFilledModal.tsx:67-81 ordering rationale —
  --    hired applicant must be excluded from the notify-job-filled NOTIFY_STATUSES sweep
  --    that fires from the on_job_filled trigger on the UPDATE jobs below.)
  IF p_applicant_id IS NOT NULL THEN
    UPDATE public.applications
       SET status = 'hired'
     WHERE id = p_applicant_id
       AND job_id = p_job_id;
    -- Defensive: applicant must belong to this job. If not, no rows updated; raise.
    IF NOT FOUND THEN
      RAISE EXCEPTION 'Applicant % not found for job %', p_applicant_id, p_job_id
        USING ERRCODE = 'P0002';
    END IF;
  END IF;

  -- 5. Mark job filled — fires on_job_filled trigger inside the same transaction.
  --    pg_net.http_post is async (queued in net._http_response); the trigger does not
  --    block the UPDATE on Edge fn response. If the trigger function itself raises,
  --    the entire RPC rolls back (steps 4 + 5 atomic).
  UPDATE public.jobs
     SET status = 'filled'
   WHERE id = p_job_id;

  IF NOT FOUND THEN
    RAISE EXCEPTION 'Failed to mark job % filled', p_job_id USING ERRCODE = 'P0001';
  END IF;
END;
$$;

GRANT EXECUTE ON FUNCTION public.mark_job_filled(uuid, uuid) TO authenticated;

COMMIT;
```

### Pattern 2: AFTER UPDATE trigger w/ OLD/NEW guard

**What:** Side-effect on a status transition. Trigger function fires only when the guard matches (status was active, is no longer).

**Example skeleton (full body — paste-ready):**

```sql
-- Source: derived from supabase/migrations/017_notify_job_filled_webhook.sql
-- (AFTER UPDATE trigger + OLD/NEW guard + SECURITY DEFINER).
-- 027_match_scores_cleanup_trigger.sql

BEGIN;

-- 1. Trigger function
CREATE OR REPLACE FUNCTION public.cleanup_match_scores_on_status_change()
RETURNS trigger
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
BEGIN
  -- Guard: fire only when the job leaves 'active' status.
  -- Active → non-active (filled / expired / archived) sheds match_scores.
  -- Non-active → active reactivation: match_scores get recomputed on demand
  -- by existing engine (compute_match_scores_batch), no special handling here.
  IF OLD.status = 'active' AND NEW.status IS DISTINCT FROM 'active' THEN
    DELETE FROM public.match_scores WHERE job_id = NEW.id;
    -- match_scores has match_scores_job_id_idx (001_initial_schema.sql:187),
    -- so the DELETE is index-backed.
  END IF;
  RETURN NEW;
END;
$$;

-- 2. Attach trigger. Distinct trigger name from on_job_filled (017) so both fire
--    on AFTER UPDATE jobs without conflict. Trigger order is defined by name —
--    Postgres fires alphabetically; on_job_filled fires before
--    on_jobs_status_change_match_scores_cleanup, both inside the same transaction.
DROP TRIGGER IF EXISTS on_jobs_status_change_match_scores_cleanup ON public.jobs;
CREATE TRIGGER on_jobs_status_change_match_scores_cleanup
  AFTER UPDATE OF status ON public.jobs
  FOR EACH ROW
  EXECUTE FUNCTION public.cleanup_match_scores_on_status_change();

-- 3. One-time backfill — clears stale rows that accumulated pre-trigger.
DELETE FROM public.match_scores ms
 WHERE EXISTS (
   SELECT 1 FROM public.jobs j
    WHERE j.id = ms.job_id
      AND j.status IS DISTINCT FROM 'active'
 );

-- 4. Post-verify
DO $verify$
DECLARE
  v_orphan_count int;
  v_trigger_count int;
BEGIN
  SELECT COUNT(*) INTO v_orphan_count
    FROM public.match_scores ms
    JOIN public.jobs j ON j.id = ms.job_id
   WHERE j.status IS DISTINCT FROM 'active';
  IF v_orphan_count > 0 THEN
    RAISE EXCEPTION 'match_scores backfill failed: % orphans remain', v_orphan_count;
  END IF;

  SELECT COUNT(*) INTO v_trigger_count
    FROM pg_trigger
   WHERE tgname = 'on_jobs_status_change_match_scores_cleanup'
     AND NOT tgisinternal;
  IF v_trigger_count <> 1 THEN
    RAISE EXCEPTION 'Trigger not attached: count=%', v_trigger_count;
  END IF;
  RAISE NOTICE 'Post-verify OK: zero orphans, trigger attached.';
END;
$verify$;

COMMIT;
```

**Note on `crypto.timingSafeEqual`:** CONTEXT.md mentioned this as a planner-discretion call. The `get-resend-stats` precedent uses plain `!==` string comparison (line 47). If hardening is desired in 18.2, retrofit both fns at once.

### Pattern 4: pg_net `X-Webhook-Secret` header injection (SQL caller side)

**What:** The two existing trigger functions (`handle_job_filled` from `017`/`022`, and the placement-followup cron job from `011_placement_fee_followups.sql:58-80`) need a new header `x-webhook-secret`. Use `current_setting('app.webhook_secret', true)` so the SQL doesn't bake the literal — operator sets it via `ALTER DATABASE postgres SET app.webhook_secret = '<value>';`.

**Strategy: NEW migration `028_*` that ALTERs both** (cleaner audit trail than editing 017/022/011 in place; matches the 022-amends-017 precedent).

**Example skeleton (paste-ready):**

```sql
-- Source: derived from 022_fix_pg_net_http_post_signature.sql
-- (CREATE OR REPLACE FUNCTION strategy: function body change picks up automatically;
--  trigger remains intact since it points at the function name.)
-- 028_webhook_secret_pg_net_callers.sql

BEGIN;

-- ============================================================
-- Part 1 — handle_job_filled (notify-job-filled trigger caller)
--   Replaces the body from 022 with one extra header line.
-- ============================================================

CREATE OR REPLACE FUNCTION public.handle_job_filled()
RETURNS trigger
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  payload jsonb;
  v_secret text;
BEGIN
  IF (OLD.status IS DISTINCT FROM 'filled') AND (NEW.status = 'filled') THEN
    payload := jsonb_build_object(
      'type', 'UPDATE',
      'table', 'jobs',
      'schema', 'public',
      'record', row_to_json(NEW)::jsonb,
      'old_record', row_to_json(OLD)::jsonb
    );

    -- Read app.webhook_secret from DB-level config. true = silent NULL on missing
    -- so the trigger doesn't error if operator hasn't set the secret yet
    -- (defence-in-depth: Edge fn returns 403 on null, no harm done).
    v_secret := current_setting('app.webhook_secret', true);

    PERFORM net.http_post(
      url := (SELECT decrypted_secret FROM vault.decrypted_secrets WHERE name = 'supabase_url') || '/functions/v1/notify-job-filled',
      headers := jsonb_build_object(
        'Content-Type',     'application/json',
```

### Pattern 5: 15× FK index migration

**What:** Plain `CREATE INDEX IF NOT EXISTS <table>_<col>_idx` lines, one per advisor lint. Naming follows `001_initial_schema.sql:140-188` precedent (`<table>_<col>_idx`).

**`CREATE INDEX` vs `CREATE INDEX CONCURRENTLY` decision:** Studio SQL Editor wraps statements in an implicit transaction by default; `CREATE INDEX CONCURRENTLY` cannot run inside a transaction. Plain `CREATE INDEX` is online-safe (acquires `SHARE` lock — blocks writes but allows reads). Pre-launch tables are tiny — `admin_notes`, `placement_fees`, `message_threads`, `messages` (Growth Phase tables, currently empty), `listing_fees` (Phase 5 — single-digit rows). Index build is ms-scale on these. Recommend plain `CREATE INDEX IF NOT EXISTS` (matches every other index in the repo, no per-statement transaction breakout).

### Anti-Patterns to Avoid

- **Editing migrations 017 / 022 / 011 in place** to add the X-Webhook-Secret header. Migrations are append-only history — write `028_*` that uses `CREATE OR REPLACE FUNCTION` instead. Matches the `022 amends 017` precedent (022 didn't edit 017 — it shipped a new migration).
- **Hand-rolled JWT validation** in the Edge fns. CLAUDE §5 explains why we don't go down that path. The `WEBHOOK_SECRET` route restores defence-in-depth without re-engaging the legacy JWT incompatibility.
- **`crypto.subtle.timingSafeEqual` divergence** from the `get-resend-stats` precedent. Match the existing pattern; if hardening becomes desirable, retrofit all three Edge fns in 18.2 together.
- **`CREATE INDEX CONCURRENTLY` inside Studio SQL Editor** — fails because Studio wraps multi-statement runs in a transaction.
- **Letting MarkFilledModal's two PostgREST UPDATEs survive after the RPC ships.** Single-RPC contract is the whole point — leaving the two `.update()` calls in as fallback re-introduces the orphan-hired class of bug.

## Don't Hand-Roll

| Problem | Don't build | Use instead | Why |
|---------|-------------|-------------|-----|
| Multi-statement transactional logic in client | Manual try/catch + compensation rollback in `MarkFilledModal.tsx` | SECURITY DEFINER RPC with implicit transaction | Compensation rollback in client code is non-atomic by definition — if the rollback itself fails, you're worse off. |
| Webhook auth across 2 Edge fns | New custom auth scheme | Replicate `get-resend-stats:27,43-49` byte-for-byte | We already have a production-precedent for this exact problem. Diverging means two auth code-paths in the codebase, not one. |
| Cleanup background job for stale match_scores | Cron job + cleanup script + scheduling infra | AFTER UPDATE trigger w/ OLD/NEW guard | Trigger fires synchronously inside the transition transaction. Zero job-scheduling surface, zero "what if cron didn't run." |
| Doc-types drift detection | CI bash regex script | vitest static-source test | Already proven pattern (17-04). Local `pnpm test` catches it pre-push. CI runs the same `pnpm test` in `supabase-deploy.yml`. |
| Bespoke FK index migration tool | Read advisor JSON + generate at runtime | Hand-paste 15 lines | 15 is below the threshold where automation is worth the surface area. The advisor list is stable; copy-paste once, audit on review. |

**Key insight:** Phase 18.1 is overwhelmingly an *assembly* phase. Every primitive has a precedent. Hand-rolling is a code smell here.

## Common Pitfalls

### Pitfall 1: Operator gates code without setting `WEBHOOK_SECRET` — production breaks silently

**What goes wrong:** Wave 2 lands the Edge fn validation block. Operator hasn't run `supabase secrets set WEBHOOK_SECRET=...`. Every `notify-job-filled` invocation returns 503 ("Server misconfigured"). MAIL-02 silently fails (no emails sent on job-fill).

**Why it happens:** Wave 2 (code) and Wave 3 (operator) are sequenced — but the gap between them is real elapsed clock time. If anything fills a job in that window, the trigger fires, pg_net posts to `notify-job-filled`, Edge fn 503s, no email.

**How to avoid:** Operator-action plan must run secret-setting BEFORE Edge fn redeploy. Specifically:

1. `supabase secrets set WEBHOOK_SECRET=<hex>` (operator action; secret now set in Edge fn runtime env)
2. Studio SQL: `ALTER DATABASE postgres SET app.webhook_secret = '<same hex>';` (DB-side setting now matches)
3. THEN `supabase functions deploy notify-job-filled send-followup-emails` (Edge fn picks up secret on first cold start)
4. THEN run synthetic UAT (curl with header → expect 200; curl without header → expect 403)

The 503 fallback (`if (!WEBHOOK_SECRET)`) is a backstop, not a workaround — operator-action plan must finish all 4 steps in one window.

**Warning signs:** `notify-job-filled` Edge fn logs showing `WEBHOOK_SECRET unset — refusing to process` after Wave 2 deploy; 503 status on synthetic curl.

### Pitfall 2: `mark_job_filled` RPC's UPDATE fires `on_job_filled` trigger; if trigger raises (e.g., pg_net unavailable), entire RPC rolls back

**What goes wrong:** Trigger function `handle_job_filled` calls `pg_net.http_post`. If the call raises (e.g., vault.decrypted_secrets misconfigured, pg_net signature drift like the 022 incident, or our new `028` migration introduces a typo), the trigger rolls back, the `UPDATE jobs` rolls back, the `UPDATE applications` rolls back — entire RPC errors back to client. Visible failure (good), but harder to diagnose than "email didn't arrive."

**Why it happens:** This is correct behaviour by design (CONTEXT.md "Failure mode: trigger error rolls back the jobs.status UPDATE. Acceptable") — but the planner needs to verify the Wave 1 RPC ships AFTER 022 (which already shipped 2026-05-03). Verify via read-only MCP `pg_proc` query that `handle_job_filled` body matches 022 (i.e., `body := payload`, not `body := payload::text`) before assuming a clean baseline.

**How to avoid:**
- Wave 0 verification: read `handle_job_filled` body via MCP `execute_sql` (`SELECT pg_get_functiondef('public.handle_job_filled'::regproc);`) and confirm 022's signature is live.
- Plan 028 (X-Webhook-Secret SQL caller) ships AFTER plan 026 (mark_job_filled RPC) so the RPC's UPDATE fires the 022-corrected `handle_job_filled`. Mid-Wave-2 028 then `CREATE OR REPLACE FUNCTION`s `handle_job_filled` with the new header — no RPC change needed.
- After 028 ships: post-verify that `handle_job_filled` body now contains `'X-Webhook-Secret'` — the planned `DO $verify$ block` does this.

**Warning signs:** RPC errors with code 'P0001' / 'XX000' on otherwise-valid `mark_job_filled` calls — diagnose via Supabase logs (`get_logs --service=postgres`) before retrying.

### Pitfall 3: Migration 028 vs operator timing — `app.webhook_secret` not set when migration runs

**What goes wrong:** Operator applies 028 in Studio SQL Editor (Wave 2). At this moment, `current_setting('app.webhook_secret', true)` returns NULL (operator hasn't run `ALTER DATABASE postgres SET app.webhook_secret`). The trigger function body now reads `COALESCE(v_secret, '')` → the next pg_net call sends `X-Webhook-Secret: ''`. Edge fn 403s.

**Why it happens:** Cross-boundary deploy ordering.

**How to avoid:** The `true` second arg to `current_setting` is the load-bearing safeguard — without it, the trigger would error and roll back the `UPDATE jobs`. With it, the trigger sends an empty-string header, the Edge fn returns 403, the trigger function does NOT raise (pg_net.http_post is async; the 403 is the *response*, not a SQL exception). Net effect: silent 403, no rollback, no email — same failure mode as Pitfall 1.

The operator-action plan UAT step covers this: synthetic curl post-deploy. If the synthetic call returns 403 from inside the function (caller-side empty header), the operator hasn't set `app.webhook_secret` yet. Run `ALTER DATABASE postgres SET app.webhook_secret = '<hex>'`, retry curl, expect 200.

**Warning signs:** Edge fn logs `forbidden` 403s post-deploy where curl with explicit header succeeds. Confirms DB-side `app.webhook_secret` is null.

### Pitfall 4: Match_scores cleanup trigger blocks `UPDATE jobs` if match_scores RLS or DELETE constraint errors

**What goes wrong:** Imagined: trigger fires `DELETE FROM match_scores WHERE job_id = NEW.id`; some FK constraint or RLS policy raises; entire `UPDATE jobs` rolls back. Same blast radius as Pitfall 2.

**Why it happens:** SECURITY DEFINER on the trigger function bypasses RLS, so RLS isn't a concern. But if `match_scores` has any inbound FK referencing it (it doesn't — verify via `pg_constraint`), DELETE could cascade or raise.

**How to avoid:**
- Pre-Wave-1 verification: read `pg_constraint` to confirm no FKs reference `match_scores.id`. (`match_scores` is a leaf table — score data only flows in.)
- Trigger function uses `SECURITY DEFINER` so RLS doesn't apply.
- Pre-deploy backfill DELETE in 027's body: catches stale rows that already exist. Post-verify asserts zero orphans (`pg_get_functiondef` + manual orphan-count query).

**Warning signs:** `UPDATE jobs` failing on previously-working code paths post-027 deploy. Run `SELECT pg_get_functiondef('public.cleanup_match_scores_on_status_change'::regproc);` to confirm function body matches the 027 source.

### Pitfall 5: Doc-types drift test extracts wrong literal because regex misses multiline policy body

**What goes wrong:** Vitest test's regex grabs `'cv', 'certificate', 'reference'` from `domain.ts` (single-line) and `index.ts` (single-line) but *misses* the SQL policy in `020_*.sql:35` because the policy body spans multiple lines (`document_type IN ('cv', 'certificate', 'reference')` on one line — actually a single line in this case, but the surrounding context is multiline). False positives in the regex pull in unrelated string literals.

**Why it happens:** Lazy regex like `/'cv'.*'reference'/` is fragile.

**How to avoid:** Use targeted single-line regex that anchors on the surrounding context:
- For TS: `/EMPLOYER_VISIBLE_DOCUMENT_TYPES[^=]*=\s*\[([^\]]+)\]/` then split on `,` and trim quotes.
- For Edge fn: same regex (the variable name `EMPLOYER_VISIBLE_DOCUMENT_TYPES` is repeated).
- For SQL: `/document_type IN \(([^)]+)\)/` (the `IN (...)` clause is single-line in `020:35`).
- Normalise each result into `Set<string>` and assert all three sets are equal.

Skeleton in §"Code Examples" Item #1 below shows the verbatim approach.

### Pitfall 6: 15 vs 14 FK count drift in ROADMAP

**What goes wrong:** ROADMAP §"Phase 18.1" success criterion #6 says "14× `unindexed_foreign_keys`". Live advisor count is 15 (per CONTEXT.md). The plan should reference 15 in acceptance criteria; ROADMAP is stale and should be reconciled in 18.1's verifier pass (not in the plan body — that's documentation cleanup).

**Why it happens:** ROADMAP was written before the advisor was re-run; advisor surface drifted between scope-write and plan-write.

**How to avoid:** Plan acceptance criterion: "15 indexes created, all named `<table>_<col>_idx`, verified via `pg_indexes` query post-apply." Verifier flags ROADMAP discrepancy as a docs-only fix-up — does NOT block the phase.

### Pitfall 7: Stripe webhook event audit must match what the deployed function actually handles, not what we wish it handled

**What goes wrong:** Operator UAT confirms Stripe Dashboard has `payment_intent.succeeded` registered. But the deployed Edge fn also branches on `invoice.payment_succeeded` (line 128). If Dashboard doesn't list `invoice.payment_succeeded`, the placement-fee invoice flow silently breaks on first real charge.

**Why it happens:** Dashboard event registration is operator memory; code branches grow over time.

**How to avoid:** Audit `supabase/functions/stripe-webhook/index.ts` event branches BEFORE writing UAT script. Confirmed via grep (`grep -n "event.type" supabase/functions/stripe-webhook/index.ts`):

| Line | Event type the function handles |
|------|----------------------------------|
| 48 | `payment_intent.succeeded` |
| 128 | `invoice.payment_succeeded` |

UAT script must list BOTH events as required dashboard registrations. Also verify `STRIPE_WEBHOOK_SECRET` is set (separate from `STRIPE_SECRET_KEY` — Edge fn line 16). Both secrets must be present and live-mode-paired.

**Warning signs:** First real placement charge succeeds in Stripe but `placement_fees.stripe_invoice_id` never updates. Diagnose via `supabase functions logs stripe-webhook` for missing event branch.

### Item #1 — Doc-types canonical-source vitest drift test

**File:** `tests/employer-visible-document-types-drift.test.ts`

```typescript
// Phase 18.1 #1 — EMPLOYER_VISIBLE_DOCUMENT_TYPES drift guard.
//
// The list MUST be identical across three layers:
//   1. src/types/domain.ts:200      (TS canonical)
//   2. supabase/functions/get-applicant-document-url/index.ts:43 (Edge fn whitelist)
//   3. supabase/migrations/020_seeker_documents_employer_policy.sql:35 (RLS policy)
//
// Drift means an employer either gets denied access to a doc they should see
// (false negative) OR an identity doc leaks (false positive — privacy regression
// on BFIX-02 / PRIV-02). Either way, prod-blocker.
//
// Pattern source: tests/saved-search-load-integration.test.tsx (Phase 17-04 static
// source test — readFileSync + regex; pure-Node, no jsdom, runs in <5ms).

import { describe, it, expect } from 'vitest'
import { readFileSync } from 'node:fs'
import { resolve } from 'node:path'

const ROOT = resolve(__dirname, '..')
const TS_PATH    = resolve(ROOT, 'src/types/domain.ts')
const EDGE_PATH  = resolve(ROOT, 'supabase/functions/get-applicant-document-url/index.ts')
const SQL_PATH   = resolve(ROOT, 'supabase/migrations/020_seeker_documents_employer_policy.sql')

function extractTsList(source: string): string[] {
  // Match: EMPLOYER_VISIBLE_DOCUMENT_TYPES[: DocumentType[]]? = ['cv', 'certificate', 'reference']
  const m = source.match(/EMPLOYER_VISIBLE_DOCUMENT_TYPES[^=]*=\s*\[([^\]]+)\]/)
  if (!m) throw new Error('TS: EMPLOYER_VISIBLE_DOCUMENT_TYPES literal not found')
  return m[1].split(',').map((s) => s.trim().replace(/^['"]|['"]$/g, ''))
}

function extractSqlList(source: string): string[] {
  // Match: document_type IN ('cv', 'certificate', 'reference')
  const m = source.match(/document_type\s+IN\s*\(([^)]+)\)/)
  if (!m) throw new Error('SQL: document_type IN (...) literal not found')
  return m[1].split(',').map((s) => s.trim().replace(/^'|'$/g, ''))
}

describe('EMPLOYER_VISIBLE_DOCUMENT_TYPES drift guard', () => {
  it('TS canonical, Edge fn duplicate, and SQL policy literal are identical', () => {
    const tsList   = extractTsList(readFileSync(TS_PATH,   'utf8'))
    const edgeList = extractTsList(readFileSync(EDGE_PATH, 'utf8'))
    const sqlList  = extractSqlList(readFileSync(SQL_PATH, 'utf8'))

    // Order-independent comparison via sorted sets
    const sorted = (xs: string[]) => [...new Set(xs)].sort()
    expect(sorted(tsList)).toEqual(sorted(edgeList))
    expect(sorted(tsList)).toEqual(sorted(sqlList))
  })

  it('canonical list does NOT contain "identity" (privacy regression guard)', () => {
    const tsList = extractTsList(readFileSync(TS_PATH, 'utf8'))
    expect(tsList).not.toContain('identity')
  })
})
```
