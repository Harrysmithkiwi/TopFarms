```typescript
import { useEffect, useState } from 'react'
import { supabase } from '@/lib/supabase'
import { useAuth } from '@/hooks/useAuth'

/**
 * Returns the current seeker's seeker_profiles.id.
 *
 * - Returns null while loading or when the user is not signed in
 * - Returns null if the user is signed in but has no seeker_profile row
 *   (e.g. employer-role users)
 * - Memoised on session.user.id — only re-fetches when auth changes
 */
export function useSeekerProfileId(): string | null {
  const { session } = useAuth()
  const [seekerProfileId, setSeekerProfileId] = useState<string | null>(null)

  useEffect(() => {
    if (!session?.user?.id) {
      setSeekerProfileId(null)
      return
    }
    let cancelled = false
    supabase
      .from('seeker_profiles')
      .select('id')
      .eq('user_id', session.user.id)
      .maybeSingle()
      .then(({ data, error }) => {
        if (cancelled) return
        if (error) {
          console.error('useSeekerProfileId: lookup failed', error)
          setSeekerProfileId(null)
          return
        }
        setSeekerProfileId(data?.id ?? null)
      })
    return () => { cancelled = true }
  }, [session?.user?.id])

  return seekerProfileId
}
```

Key requirements:
- Single-shot lookup on session change (cached in component state)
- `let cancelled = false` pattern to avoid setState after unmount during the async resolve
- `maybeSingle()` (not `single()`) so an absent row returns `data: null` instead of erroring
- No dependence on the seeker's role — caller decides what to do with `null`

Create `src/hooks/useAppliedStatuses.ts` — single batch query against `applications`, returns the latest status per job.

```typescript
import { useEffect, useState } from 'react'
import { supabase } from '@/lib/supabase'
import type { ApplicationStatus } from '@/types/domain'

/**
 * Batches a single applications query for the current seeker over the
 * provided jobIds and returns Map<jobId, latestStatus>.
 *
 * - Returns an empty Map while loading, when seekerProfileId is null,
 *   or when jobIds is empty
 * - "Latest" is the most recent row per job by created_at descending
 *   (a seeker may have re-applied after a withdrawal, etc.)
 */
export function useAppliedStatuses(
  jobIds: string[],
  seekerProfileId: string | null,
): Map<string, ApplicationStatus> {
  const [statuses, setStatuses] = useState<Map<string, ApplicationStatus>>(new Map())

  // Stable join key for the effect dep array (sorted to be deterministic)
  const jobIdKey = [...jobIds].sort().join(',')

  useEffect(() => {
    if (!seekerProfileId || jobIds.length === 0) {
      setStatuses(new Map())
      return
    }
    let cancelled = false
    supabase
      .from('applications')
      .select('job_id, status, created_at')
      .eq('seeker_id', seekerProfileId)
      .in('job_id', jobIds)
      .order('created_at', { ascending: false })
      .then(({ data, error }) => {
        if (cancelled) return
        if (error) {
          console.error('useAppliedStatuses: query failed', error)
          setStatuses(new Map())
          return
        }
        // Build Map keeping the FIRST encounter per job_id (rows are
        // ordered DESC, so first = latest)
        const map = new Map<string, ApplicationStatus>()
        for (const row of data ?? []) {
          if (!map.has(row.job_id)) {
            map.set(row.job_id, row.status as ApplicationStatus)
          }
        }
        setStatuses(map)
      })
    return () => { cancelled = true }
  }, [seekerProfileId, jobIdKey])

  return statuses
}
```

Key requirements:
- Single batch query (one round-trip), not one query per job
- `ORDER BY created_at DESC` so the first row encountered per job_id is the latest
- Empty Map fallback for loading / no-seeker / no-jobs cases — caller treats absence as "not applied"
- `jobIdKey` (sorted, joined) used as the effect dep to avoid infinite re-fires when the array reference changes but contents are equivalent

-- Idempotent: re-running deletes prior UAT applications and re-creates them.

BEGIN;

-- Identify the seeker by email (adjust to your test seeker)
DO $$
DECLARE
  v_user_id uuid;
  v_seeker_id uuid;
  v_active_job uuid;
  v_terminal_job uuid;
BEGIN
  -- 1. Resolve auth user
  SELECT id INTO v_user_id FROM auth.users WHERE email = 'YOUR_TEST_SEEKER_EMAIL@example.com';
  IF v_user_id IS NULL THEN
    RAISE EXCEPTION 'No auth user found for the test email. Adjust the email above.';
  END IF;

  -- 2. Resolve seeker profile
  SELECT id INTO v_seeker_id FROM seeker_profiles WHERE user_id = v_user_id;
  IF v_seeker_id IS NULL THEN
    RAISE EXCEPTION 'No seeker_profiles row for that user. Complete onboarding first.';
  END IF;

  -- 3. Pick two arbitrary active jobs (adjust the WHERE if you want specific titles)
  SELECT id INTO v_active_job   FROM jobs WHERE status = 'active' ORDER BY created_at DESC LIMIT 1;
  SELECT id INTO v_terminal_job FROM jobs WHERE status = 'active' ORDER BY created_at DESC OFFSET 1 LIMIT 1;
  IF v_active_job IS NULL OR v_terminal_job IS NULL OR v_active_job = v_terminal_job THEN
    RAISE EXCEPTION 'Need at least two distinct active jobs in the jobs table. Create them first.';
  END IF;

  -- 4. Wipe any prior UAT applications for these (job, seeker) pairs (idempotent)
  DELETE FROM applications WHERE seeker_id = v_seeker_id AND job_id IN (v_active_job, v_terminal_job);

  -- 5. Insert one active application + one terminal (declined) application
  INSERT INTO applications (job_id, seeker_id, status, created_at)
  VALUES
    (v_active_job,   v_seeker_id, 'applied',  now() - interval '2 days'),
    (v_terminal_job, v_seeker_id, 'declined', now() - interval '5 days');

  RAISE NOTICE 'Seeded UAT for seeker_id %, active_job %, declined_job %', v_seeker_id, v_active_job, v_terminal_job;
END $$;

COMMIT;
