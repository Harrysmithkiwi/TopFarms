```sql
-- ============================================================
-- 019_seeker_documents.sql
-- TopFarms — Phase 14 BFIX-03 — document categorization
-- Sections:
--   1. seeker_documents table with document_type enum
--   2. RLS policies (seeker-only access; Edge Function uses service role)
--   3. Indexes
--   4. Idempotent LATERAL backfill from seeker_profiles.document_urls
--   5. Deprecation comment on legacy column
-- ============================================================

BEGIN;

-- ============================================================
-- 1. seeker_documents table
-- ============================================================

CREATE TABLE IF NOT EXISTS public.seeker_documents (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  seeker_id       uuid NOT NULL REFERENCES public.seeker_profiles(id) ON DELETE CASCADE,
  storage_path    text NOT NULL,
  document_type   text NOT NULL CHECK (document_type IN ('cv', 'certificate', 'reference', 'identity', 'other')),
  filename        text NOT NULL,
  uploaded_at     timestamptz NOT NULL DEFAULT now(),
  file_size_bytes bigint
);

-- ============================================================
-- 2. RLS — seeker-only direct access
-- Edge Function (14-03) uses service role and bypasses RLS
-- ============================================================

ALTER TABLE public.seeker_documents ENABLE ROW LEVEL SECURITY;

CREATE POLICY "seekers select own documents"
ON public.seeker_documents FOR SELECT
TO authenticated
USING (
  seeker_id IN (SELECT id FROM public.seeker_profiles WHERE user_id = auth.uid())
);

CREATE POLICY "seekers insert own documents"
ON public.seeker_documents FOR INSERT
TO authenticated
WITH CHECK (
  seeker_id IN (SELECT id FROM public.seeker_profiles WHERE user_id = auth.uid())
);

CREATE POLICY "seekers update own documents"
ON public.seeker_documents FOR UPDATE
TO authenticated
USING (
  seeker_id IN (SELECT id FROM public.seeker_profiles WHERE user_id = auth.uid())
)
WITH CHECK (
  seeker_id IN (SELECT id FROM public.seeker_profiles WHERE user_id = auth.uid())
);

CREATE POLICY "seekers delete own documents"
ON public.seeker_documents FOR DELETE
TO authenticated
USING (
  seeker_id IN (SELECT id FROM public.seeker_profiles WHERE user_id = auth.uid())
);

-- ============================================================
-- 3. Indexes
-- ============================================================

CREATE INDEX IF NOT EXISTS seeker_documents_seeker_id_idx
  ON public.seeker_documents(seeker_id);

CREATE INDEX IF NOT EXISTS seeker_documents_seeker_id_document_type_idx
  ON public.seeker_documents(seeker_id, document_type);

-- ============================================================
-- 4. Idempotent LATERAL backfill from seeker_profiles.document_urls
-- Existing storage paths are migrated as document_type='other'
-- The NOT EXISTS guard makes the migration safe to re-run
-- ============================================================

INSERT INTO public.seeker_documents (seeker_id, storage_path, document_type, filename)
SELECT sp.id, doc.path, 'other', regexp_replace(doc.path, '^.*/', '')
FROM public.seeker_profiles sp,
     LATERAL unnest(sp.document_urls) AS doc(path)
WHERE sp.document_urls IS NOT NULL
  AND array_length(sp.document_urls, 1) > 0
  AND NOT EXISTS (
    SELECT 1 FROM public.seeker_documents sd
    WHERE sd.storage_path = doc.path AND sd.seeker_id = sp.id
  );

-- ============================================================
-- 5. Deprecation marker on legacy column
-- Column is preserved for now; drop in a follow-up cleanup phase
-- ============================================================

COMMENT ON COLUMN public.seeker_profiles.document_urls IS
  'DEPRECATED: replaced by seeker_documents table from migration 019 (Phase 14 BFIX-03). New uploads write to seeker_documents only. Drop this column in a future cleanup phase once all readers have been migrated.';

COMMIT;
```

Key requirements:
- Transaction wrapper so partial failure rolls back cleanly

- RLS uses `seeker_profiles.user_id = auth.uid()` subquery — same pattern as `seeker_contacts` and other seeker-scoped tables

```typescript
import { useState } from 'react'
import { FileDropzone } from '@/components/ui/FileDropzone'
import { Select } from '@/components/ui/Select'
import { supabase } from '@/lib/supabase'
import type { DocumentType, SeekerDocument } from '@/types/domain'
import { DOCUMENT_TYPE_LABELS } from '@/types/domain'

interface DocumentUploaderProps {
  seekerProfileId: string
  bucket: string
  path: string
  accept: Record<string, string[]>
  maxSize?: number
  maxFiles?: number
  /** Called after each row is inserted into seeker_documents */
  onDocumentInserted?: (doc: SeekerDocument) => void
}

const TYPE_OPTIONS = (Object.keys(DOCUMENT_TYPE_LABELS) as DocumentType[]).map((value) => ({
  value,
  label: DOCUMENT_TYPE_LABELS[value],
}))

export function DocumentUploader({
  seekerProfileId,
  bucket,
  path,
  accept,
  maxSize,
  maxFiles,
  onDocumentInserted,
}: DocumentUploaderProps) {
  // Map keyed by the storage path that FileDropzone produces; value is the
  // document_type the seeker selected for that file
  const [typesByPath, setTypesByPath] = useState<Record<string, DocumentType>>({})

  async function handleUploadComplete(uploadedPath: string, file: File) {
    const docType = typesByPath[uploadedPath]
    if (!docType) {
      console.warn('DocumentUploader: upload completed for path without a type — refusing to insert')
      return
    }
    const filename = file.name
    const { data, error } = await supabase
      .from('seeker_documents')
      .insert({
        seeker_id: seekerProfileId,
        storage_path: uploadedPath,
        document_type: docType,
        filename,
        file_size_bytes: file.size,
      })
      .select()
      .single()
    if (error) {
      console.error('DocumentUploader: seeker_documents insert failed', error)
      return
    }
    onDocumentInserted?.(data as SeekerDocument)
  }

  return (
    <div className="flex flex-col gap-3">
      <FileDropzone
        bucket={bucket}
        path={path}
        accept={accept}
        maxSize={maxSize}
        maxFiles={maxFiles}
        privateMode
        // Per-file type-picker is rendered via FileDropzone's row-render extension
        // (or via the renderFileRow prop if available — otherwise wrap externally)
        // For each file in the staging list, render:
        //   <Select options={TYPE_OPTIONS} value={typesByPath[file.name]}
        //           placeholder="Document type" onValueChange={...} />
        // Upload button stays disabled until every staged file has typesByPath[file.name] !== undefined.
        onUploadComplete={handleUploadComplete}
      />
      <p className="text-xs text-mid">
        Each file requires a document type before upload. Identity documents are
        kept private — employers will never see them.
      </p>
    </div>
  )
}
```

```typescript
import { useEffect, useState } from 'react'
import { toast } from 'sonner'
import { Nav } from '@/components/layout/Nav'
import { Select } from '@/components/ui/Select'
import { Button } from '@/components/ui/Button'
import { supabase } from '@/lib/supabase'
import { useSeekerProfileId } from '@/hooks/useSeekerProfileId'
import { DOCUMENT_TYPE_LABELS } from '@/types/domain'
import type { DocumentType, SeekerDocument } from '@/types/domain'

const TYPE_OPTIONS = (Object.keys(DOCUMENT_TYPE_LABELS) as DocumentType[]).map((value) => ({
  value,
  label: DOCUMENT_TYPE_LABELS[value],
}))

export function SeekerDocuments() {
  const seekerProfileId = useSeekerProfileId()
  const [docs, setDocs] = useState<SeekerDocument[] | null>(null)
  const [loading, setLoading] = useState(true)

  async function loadDocs() {
    if (!seekerProfileId) return
    setLoading(true)
    const { data, error } = await supabase
      .from('seeker_documents')
      .select('*')
      .eq('seeker_id', seekerProfileId)
      .order('uploaded_at', { ascending: false })
    setLoading(false)
    if (error) {
      toast.error('Failed to load documents')
      return
    }
    setDocs((data ?? []) as SeekerDocument[])
  }

  useEffect(() => { loadDocs() }, [seekerProfileId])

  async function handleTypeChange(docId: string, newType: DocumentType) {
    const { error } = await supabase
      .from('seeker_documents')
      .update({ document_type: newType })
      .eq('id', docId)
    if (error) {
      toast.error('Failed to update document type')
      return
    }
    toast.success('Document type updated')
    await loadDocs()
  }

  async function handleDelete(doc: SeekerDocument) {
    if (!confirm(`Delete ${doc.filename}? This cannot be undone.`)) return
    const { error: storageError } = await supabase.storage
      .from('seeker-documents')
      .remove([doc.storage_path])
    if (storageError) {
      toast.error('Failed to delete file from storage')
      return
    }
    const { error: rowError } = await supabase
      .from('seeker_documents')
      .delete()
      .eq('id', doc.id)
    if (rowError) {
      toast.error('Storage file deleted but row removal failed — refresh to verify')
      return
    }
    toast.success('Document deleted')
    await loadDocs()
  }

  return (
    <div className="min-h-screen" style={{ backgroundColor: 'var(--color-cream)' }}>
      <Nav />
      <main className="max-w-3xl mx-auto px-4 py-10">
        <h1 className="font-display text-3xl font-bold mb-2" style={{ color: 'var(--color-soil)' }}>
          My Documents
        </h1>
        <p className="text-mid mb-6 text-sm">
          Manage your uploaded documents. Identity documents are kept private and
          are never shown to employers.
        </p>

        {loading && <p className="text-mid">Loading…</p>}

        {!loading && docs && docs.length === 0 && (
          <p className="text-mid">
            No documents uploaded yet. Add some during onboarding or from your profile.
          </p>
        )}

        {!loading && docs && docs.length > 0 && (
          <div className="flex flex-col gap-3">
            {docs.map((doc) => (
              <div
                key={doc.id}
                className="bg-white border border-fog rounded-[12px] p-4 flex items-center gap-4"
              >
                <div className="flex-1 min-w-0">
                  <p className="font-body font-semibold text-soil truncate">{doc.filename}</p>
                  <p className="text-xs text-mid">
                    Uploaded {new Date(doc.uploaded_at).toLocaleDateString()}
                    {doc.file_size_bytes ? ` · ${Math.round(doc.file_size_bytes / 1024)} KB` : ''}
                  </p>
                </div>
                <Select
                  value={doc.document_type}
                  onValueChange={(val) => handleTypeChange(doc.id, val as DocumentType)}
                  options={TYPE_OPTIONS}
                />
                <Button variant="ghost" size="sm" onClick={() => handleDelete(doc)}>
                  Delete
                </Button>
              </div>
            ))}
          </div>
        )}
      </main>
    </div>
  )
}
```
