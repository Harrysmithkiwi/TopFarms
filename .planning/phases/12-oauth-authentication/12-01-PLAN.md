```typescript
const signInWithOAuth = async (provider: 'google' | 'facebook') => {
  const { error } = await supabase.auth.signInWithOAuth({
    provider,
    options: {
      redirectTo: `${window.location.origin}/auth/select-role`,
      scopes: provider === 'facebook' ? 'email' : undefined,
    },
  })
  if (error) throw error
}
```
- `redirectTo` set to `/auth/select-role` so new OAuth users land on role selection
- Facebook requires `scopes: 'email'` to get the user's email address (per Pitfall 6 in RESEARCH.md)
- Google does not need extra scopes — `openid`, `email`, `profile` are default
- Throws error so calling component can catch and show toast

const refreshRole = async () => {

  return null
}
