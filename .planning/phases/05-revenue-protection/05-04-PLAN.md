```typescript
const RESEND_API_KEY = Deno.env.get('RESEND_API_KEY')
const FROM_EMAIL = Deno.env.get('RESEND_FROM_EMAIL') ?? 'TopFarms <onboarding@resend.dev>'

async function sendEmail(to: string, subject: string, html: string) {
  const res = await fetch('https://api.resend.com/emails', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      Authorization: `Bearer ${RESEND_API_KEY}`,
    },
    body: JSON.stringify({ from: FROM_EMAIL, to: [to], subject, html }),
  })
  if (!res.ok) {
    const error = await res.text()
    console.error(`Resend error: ${error}`)
    return false
  }
  return true
}
```

Simpler approach — query placement_fees with manual lookups:

```typescript
const { data: dueRows } = await supabaseClient
  .from('placement_fees')
  .select('id, application_id, job_id, employer_id, seeker_id')
  .eq('followup_7d_due', true)
  .eq('followup_7d_sent', false)
  .is('confirmed_at', null)

for (const row of dueRows ?? []) {
  // Get job title + employer details
  const { data: job } = await supabaseClient
    .from('jobs')
    .select('title')
    .eq('id', row.job_id)
    .single()

  const { data: employer } = await supabaseClient
    .from('employer_profiles')
    .select('farm_name, user_id')
    .eq('id', row.employer_id)
    .single()

  const { data: seeker } = await supabaseClient
    .from('seeker_profiles')
    .select('region, user_id')
    .eq('id', row.seeker_id)
    .single()

  // Get emails from auth.users via admin API
  // Actually, for employer email use session... No, this is server-side.
  // Use seeker_contacts for seeker email, and employer_profiles user_id to look up auth user email.

  // For employer email: query auth.users is not available via Supabase client.
  // Alternative: store employer email on employer_profiles, or use a service role admin call.
  // Simplest: use supabaseClient.auth.admin.getUserById(employer.user_id)
  const { data: { user: empUser } } = await supabaseClient.auth.admin.getUserById(employer?.user_id ?? '')
  const employerEmail = empUser?.email

  // For seeker email: query seeker_contacts (service role bypasses RLS)
  const { data: seekerContact } = await supabaseClient
    .from('seeker_contacts')
    .select('email')
    .eq('user_id', seeker?.user_id ?? '')
    .maybeSingle()
  const seekerEmail = seekerContact?.email

  const seekerName = seeker?.region ? `Seeker from ${seeker.region}` : 'Your shortlisted candidate'
  const farmName = employer?.farm_name ?? 'Your farm'
  const jobTitle = job?.title ?? 'your position'

  // Send employer email
  if (employerEmail) {
    await sendEmail(
      employerEmail,
      `How's it going with ${seekerName}?`,
      emailWrapper(employerDay7Body(farmName, seekerName, jobTitle, row.job_id))
    )
  }

  // Send seeker email
  if (seekerEmail) {
    await sendEmail(
      seekerEmail,
      `You were shortlisted by ${farmName}!`,
      emailWrapper(seekerDay7Body(seekerName, farmName, jobTitle))
    )
  }

  // Mark sent
  await supabaseClient
    .from('placement_fees')
    .update({ followup_7d_sent: true, followup_7d_due: false })
    .eq('id', row.id)
}
```
